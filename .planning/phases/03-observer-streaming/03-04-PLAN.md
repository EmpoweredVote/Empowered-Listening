    ```ts
    import { notFound } from 'next/navigation';
    import { getPool } from '@/lib/db/pool';
    import { ObserverShell } from './ObserverShell';

    export const dynamic = 'force-dynamic';   // observer status changes often; do not cache

    export default async function ObserverPage({
      params,
    }: {
      params: Promise<{ debateId: string }>;
    }) {
      const { debateId } = await params;
      const pool = getPool();
      const { rows } = await pool.query<{
        id: string;
        status: 'live' | 'completed' | 'scheduled' | 'cancelled';
        mux_playback_id: string | null;
        topic: string | null;
      }>(
        `SELECT id, status, mux_playback_id, topic
         FROM listening.debates
         WHERE id = $1 AND status IN ('live', 'completed', 'scheduled')`,
        [debateId],
      );
      if (rows.length === 0) return notFound();

      return (
        <ObserverShell
          debateId={debateId}
          initialStatus={rows[0].status}
          initialPlaybackId={rows[0].mux_playback_id}
          topic={rows[0].topic}
        />
      );
    }
    ```

    Why `status IN ('live', 'completed', 'scheduled')` here (vs `('live', 'completed')` in the API)?  The server component needs to render a "Waiting for debate to start" screen for scheduled debates so the URL is shareable before the debate goes live.  The ObserverShell will poll until status flips to live.  Cancelled debates still 404.

    Create `app/debates/[debateId]/ObserverShell.tsx` as a client component:

    ```tsx
    'use client';

    import { useEffect, useState } from 'react';
    import { useObserverDebateSync } from '@/hooks/useObserverDebateSync';
    import { DesktopLayout } from './DesktopLayout';

    interface Props {
      debateId: string;
      initialStatus: 'live' | 'completed' | 'scheduled';
      initialPlaybackId: string | null;
      topic: string | null;
    }

    interface StreamState {
      status: 'live' | 'completed' | 'scheduled';
      mux_playback_id: string | null;
    }

    export function ObserverShell({ debateId, initialStatus, initialPlaybackId, topic }: Props) {
      const [stream, setStream] = useState<StreamState>({ status: initialStatus, mux_playback_id: initialPlaybackId });

      // Populate zustand store with anon snapshot + realtime.
      useObserverDebateSync(debateId);

      // Poll the stream endpoint every 5 s while not yet live.  Realtime postgres_changes on debates.status
      // also updates ObserverShell via the store (below) — polling is a belt-and-suspenders fallback if the
      // debate row update is missed.
      useEffect(() => {
        if (stream.status === 'live' || stream.status === 'completed') return;
        const iv = setInterval(async () => {
          try {
            const r = await fetch(`/api/debates/${debateId}/stream`, { cache: 'no-store' });
            if (!r.ok) return;
            const data = (await r.json()) as { mux_playback_id: string | null; status: StreamState['status'] };
            setStream(data);
          } catch {
            // Ignore — next tick will retry.
          }
        }, 5000);
        return () => clearInterval(iv);
      }, [debateId, stream.status]);

      const hlsUrl =
        stream.status !== 'scheduled' && stream.mux_playback_id
          ? `https://stream.mux.com/${stream.mux_playback_id}.m3u8`
          : null;

      // Mobile placeholder: full-width video stack.  03-05 adds the real MobileLayout.tsx and modifies THIS
      // file to swap the `<div className="md:hidden">` branch to <MobileLayout …/>.  For now, render the
      // desktop layout everywhere ≥ md and a simple stacked view below md.
      return (
        <>
          {/* Desktop (md+) */}
          <div className="hidden md:block h-screen bg-slate-950 text-slate-100">
            <DesktopLayout
              hlsUrl={hlsUrl}
              status={stream.status}
              topic={topic}
            />
          </div>
          {/* Mobile fallback (< md) — replaced by 03-05 */}
          <div className="md:hidden min-h-screen bg-slate-950 text-slate-100 p-4 flex flex-col gap-4">
            <h1 className="text-lg font-semibold">{topic ?? 'Debate'}</h1>
            {hlsUrl ? (
              // 03-05 swaps this entire block out.  For now, inline HlsPlayer keeps the URL reachable on mobile.
              <div className="aspect-video w-full bg-black text-sm text-slate-400 flex items-center justify-center">
                Mobile observer layout is coming — open this page on a desktop browser for the full experience.
              </div>
            ) : (
              <div className="aspect-video w-full bg-black text-sm text-slate-400 flex items-center justify-center">
                Waiting for debate to start…
              </div>
            )}
          </div>
        </>
      );
    }
    ```

    Why this mobile-fallback shape: 03-05 will REPLACE the `<div className="md:hidden">…</div>` block with `<MobileLayout hlsUrl={hlsUrl} status={stream.status} topic={topic} />` and will add `MobileLayout.tsx` + `MobileTabs.tsx`.  Leave a `// 03-05 swaps this entire block out.` comment so the boundary is explicit.

    - Do NOT read `ev_token` anywhere in ObserverShell or DesktopLayout.  Observers are anonymous.
    - The topic is optional — display a generic fallback if null.

       ```tsx
       useEffect(() => {
         function handleKey(e: KeyboardEvent) {
           const target = e.target as HTMLElement | null;
           if (target instanceof HTMLInputElement || target instanceof HTMLTextAreaElement || target?.isContentEditable) return;

           const video = videoContainerRef.current?.querySelector('video') as HTMLVideoElement | null;

           switch (e.key) {
             case ' ':
               if (!video) return;
               e.preventDefault(); // REQUIRED — Space otherwise scrolls the page
               if (video.paused) void video.play().catch(() => {});
               else video.pause();
               break;
             case 't': case 'T':
               transcriptPanelRef.current?.focus();
               break;
             case 'n': case 'N':
               notesPanelRef.current?.focus();
               break;
             case 'f': case 'F':
               if (videoContainerRef.current?.requestFullscreen) void videoContainerRef.current.requestFullscreen().catch(() => {});
               break;
           }
         }
         document.addEventListener('keydown', handleKey);
         return () => document.removeEventListener('keydown', handleKey);
       }, []);
       ```
