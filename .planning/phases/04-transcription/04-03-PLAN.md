    ```sql
    SELECT
      te.id,
      te.speaker_id,
      te.spoken_at,
      te.debate_time_mmss,
      te.text,
      te.confidence_score
    FROM listening.transcript_entries te
    WHERE te.debate_id = $1
      AND ($2::timestamptz IS NULL OR te.spoken_at < $2)
    ORDER BY te.spoken_at ASC
    LIMIT $3
    ```

    **Part A: hooks/useTranscriptSync.ts**

    Subscribes to the Supabase Realtime broadcast channel `transcript-{debateId}`.

    IMPORTANT: Broadcast is EPHEMERAL. Any messages sent before subscription are lost.
    The correct pattern (matching Pitfall 5 in research): load DB snapshot FIRST via the GET endpoint in TranscriptPanel's useEffect, THEN subscribe to the broadcast channel. useTranscriptSync itself only handles the subscription, not the initial load.

    ```typescript
    'use client';
    import { useEffect, useCallback } from 'react';
    import { getSupabaseBrowserClient } from '@/lib/supabase/client';

    export interface FinalEntry {
      speakerId: string;
      text: string;
      spokenAt: string;         // ISO timestamp string
      debateTimeMmss: string;
    }

    export interface InterimEntry {
      speakerId: string;
      text: string;
    }

    export function useTranscriptSync(
      debateId: string,
      onFinal: (entry: FinalEntry) => void,
      onInterim: (entry: InterimEntry) => void,
    ): void {
      // Stable callbacks to avoid re-subscribing on every render
      const stableFinal = useCallback(onFinal, []);    // eslint-disable-line react-hooks/exhaustive-deps
      const stableInterim = useCallback(onInterim, []); // eslint-disable-line react-hooks/exhaustive-deps

      useEffect(() => {
        const supabase = getSupabaseBrowserClient();

        const channel = supabase
          .channel(`transcript-${debateId}`)
          .on('broadcast', { event: 'final' }, (payload) => {
            stableFinal(payload.payload as FinalEntry);
          })
          .on('broadcast', { event: 'interim' }, (payload) => {
            stableInterim(payload.payload as InterimEntry);
          })
          .subscribe();

        return () => { void supabase.removeChannel(channel); };
      }, [debateId, stableFinal, stableInterim]);
    }
    ```
