  <action>
    Create lib/transcription/deepgram-connection.ts. This class manages one Deepgram WebSocket per speaker audio track.

    Key implementation details:
    - Import: `import { createClient } from '@deepgram/sdk';` — NOT the v3 `LiveTranscriptionEvents` enum (removed in v5)
    - Event API: `connection.on('message', (data) => { if (data.type !== 'Results') return; ... })` — this is the v5 pattern
    - Audio conversion (CRITICAL): AudioFrame.data is Int16Array. Convert BEFORE sending:
      `const pcmBuffer = Buffer.from(frame.data.buffer, frame.data.byteOffset, frame.data.byteLength);`
      DO NOT send the Int16Array directly — Deepgram will reject it or return garbled results.
    - Sample rate: use `new AudioStream(track, 16000)` to resample from WebRTC's 48kHz to 16kHz (Deepgram standard)
    - Deepgram connection params: `model: 'nova-3', language: 'en', interim_results: 'true', smart_format: 'false', punctuate: 'false', filler_words: 'true', encoding: 'linear16', sample_rate: '16000', channels: '1'`
    - Confidence thresholds: `UNAVAILABLE_THRESHOLD = 0.10` (whole chunk), `INAUDIBLE_THRESHOLD = 0.20` (individual word)
    - Segment lookup: `WHERE debate_id = $1 AND status IN ('active', 'paused')` — NOT just 'active' (prep time has status 'paused')
    - Broadcast: create and immediately dispose channel (`supabase.removeChannel(channel)`) — no persistent WebSocket from worker side
    - Reconnect: exponential backoff with `Math.min(1000 * Math.pow(2, attempt), 30_000)` — cap at 30s
    - debate_time_mmss: use `computeDebateTimeMmss(this.debateActualStart, new Date())` — wall clock, NOT Deepgram word.start timestamps

    Full class structure:

    **Part A: lib/transcription/worker.ts**

    The worker joins the LiveKit room as a server-side participant using @livekit/rtc-node (NOT livekit-client — livekit-client is browser-only and will throw ReferenceError: RTCPeerConnection is not defined in Node.js).

    ```typescript
    import {
      Room, RoomEvent, TrackKind, AudioStream,
      type RemoteTrack, type RemoteParticipant,
    } from '@livekit/rtc-node';
    import { mintToken } from '@/lib/livekit/tokens';
    import { DeepgramLiveConnection } from './deepgram-connection';
    import { pool } from '@/lib/db/pool';  // or getPool() — match what's exported

    export class TranscriptionWorker {
      private room: Room;
      private connections = new Map<string, DeepgramLiveConnection>(); // speakerId → connection

      constructor(
        private readonly debateId: string,
        private readonly roomName: string,
      ) {
        this.room = new Room();
      }

      async start(): Promise<void> {
        // Fetch debate actual_start and speaker identities from DB
        const debateResult = await pool.query<{ actual_start: Date }>(
          `SELECT actual_start FROM listening.debates WHERE id = $1`,
          [this.debateId]
        );
        const actualStart = debateResult.rows[0]?.actual_start ?? new Date();

        // Map LiveKit participant identity → listening.debate_speakers.id
        const speakerResult = await pool.query<{ livekit_identity: string; id: string }>(
          `SELECT livekit_identity, id FROM listening.debate_speakers
           WHERE debate_id = $1 AND role IN ('affirmative', 'negative')`,
          [this.debateId]
        );
        const speakerMap = new Map(speakerResult.rows.map(r => [r.livekit_identity, r.id]));

        const token = await mintToken({
          identity: `transcription-worker:${this.debateId}`,
          roomName: this.roomName,
          role: 'worker',
        });

        this.room.on(RoomEvent.TrackSubscribed, async (
          track: RemoteTrack,
          _pub: unknown,
          participant: RemoteParticipant,
        ) => {
          if (track.kind !== TrackKind.KIND_AUDIO) return;
          const speakerId = speakerMap.get(participant.identity);
          if (!speakerId) return;  // moderator or worker track — skip

          const audioStream = new AudioStream(track, 16000);  // resample to 16kHz
          const conn = new DeepgramLiveConnection(
            this.debateId,
            speakerId,
            audioStream,
            actualStart,
          );
          this.connections.set(participant.identity, conn);
          await conn.start();
        });

        this.room.on(RoomEvent.TrackUnsubscribed, (
          _track: RemoteTrack,
          _pub: unknown,
          participant: RemoteParticipant,
        ) => {
          const conn = this.connections.get(participant.identity);
          if (conn) {
            conn.stop();
            this.connections.delete(participant.identity);
          }
        });

        this.room.on(RoomEvent.Disconnected, async () => {
          console.log('[worker] room disconnected — stopping all Deepgram connections');
          for (const conn of this.connections.values()) conn.stop();
          this.connections.clear();
          // Reconnect after 2s — TrackSubscribed events will fire again on rejoin
          await new Promise(r => setTimeout(r, 2000));
          if (!this.stopped) {
            await this.room.connect(process.env.LIVEKIT_URL!, token, { autoSubscribe: true });
          }
        });

        await this.room.connect(process.env.LIVEKIT_URL!, token, { autoSubscribe: true });
      }

      private stopped = false;

      async stop(): Promise<void> {
        this.stopped = true;
        for (const conn of this.connections.values()) conn.stop();
        this.connections.clear();
        await this.room.disconnect();
      }
    }
    ```
