    Open `app/globals.css`.  Immediately after `@import "tailwindcss";` (line 1) and before the `@theme` block, insert:

    ```css
    /*
     * Tailwind v4 orientation modifiers.
     *
     * v3 shipped `landscape:` and `portrait:` built-in; v4 does not.  Without these
     * @custom-variant declarations the classes silently no-op (see 03-RESEARCH.md
     * pitfall 3).  Used by app/debates/[debateId]/MobileLayout.tsx.
     */
    @custom-variant landscape (@media (orientation: landscape));
    @custom-variant portrait  (@media (orientation: portrait));
    ```

    Why this syntax: Tailwind v4 `@custom-variant` supports both a short form `@custom-variant NAME (CONDITION)` and a long block form `@custom-variant NAME { @slot; }`.  The short form is preferred here for readability.  If the short form turns out not to work in the installed Tailwind version, fall back to:

    ```css
    @custom-variant landscape {
      @media (orientation: landscape) { @slot; }
    }
    @custom-variant portrait {
      @media (orientation: portrait) { @slot; }
    }
    ```

    After editing, run `npm run build` once and confirm no CSS compilation errors.

    Do NOT remove or reorder the existing @theme block, keyframes, or any other CSS in the file.  This is an additive change.

    Signature:

    ```ts
    export interface MobileTab {
      id: string;          // stable id, e.g. 'info' | 'transcript' | 'notes'
      label: string;       // user-visible label
      content: React.ReactNode;
    }

    interface MobileTabsProps {
      tabs: MobileTab[];   // typically 3 for v1: Info, Transcript, Notes
      className?: string;
    }

    export function MobileTabs({ tabs, className = '' }: MobileTabsProps) { … }
    ```

    Implementation:

    1. `useRef<HTMLDivElement>(null)` for the scroll container.
    2. `useState<number>(0)` for active index.
    3. `scrollToTab(index)`:

       ```tsx
       function scrollToTab(index: number) {
         const el = scrollRef.current;
         if (!el) return;
         el.scrollTo({ left: index * el.offsetWidth, behavior: 'smooth' });
         setActiveIndex(index);
       }
       ```

    4. `onScroll` on the container computes active index:

       ```tsx
       onScroll={(e) => {
         const el = e.currentTarget;
         const idx = Math.round(el.scrollLeft / el.offsetWidth);
         if (idx !== activeIndex) setActiveIndex(idx);
       }}
       ```

    5. Render markup:

       ```tsx
       <div className={`flex flex-col ${className}`}>
         {/* Tab bar (bottom per UX-01 — thumb reach) */}
         <div role="tablist" className="flex border-t border-slate-800 bg-slate-900 order-2">
           {tabs.map((tab, i) => (
             <button
               key={tab.id}
               role="tab"
               aria-selected={activeIndex === i}
               aria-controls={`mobile-tab-panel-${tab.id}`}
               onClick={() => scrollToTab(i)}
               className={`flex-1 py-3 text-xs font-medium min-h-[44px] ${
                 activeIndex === i
                   ? 'text-ev-light-blue border-t-2 border-ev-light-blue'
                   : 'text-slate-400 border-t-2 border-transparent'
               }`}
             >
               {tab.label}
             </button>
           ))}
         </div>

         {/* Swipeable panels */}
         <div
           ref={scrollRef}
           onScroll={(e) => {
             const el = e.currentTarget;
             const idx = Math.round(el.scrollLeft / el.offsetWidth);
             if (idx !== activeIndex) setActiveIndex(idx);
           }}
           className="flex flex-1 overflow-x-auto snap-x snap-mandatory order-1 min-h-0 no-scrollbar"
           style={{
             scrollbarWidth: 'none',
             WebkitOverflowScrolling: 'touch',
           } as React.CSSProperties}
         >
           {tabs.map((tab) => (
             <div
               key={tab.id}
               id={`mobile-tab-panel-${tab.id}`}
               role="tabpanel"
               className="snap-start flex-shrink-0 w-full h-full overflow-y-auto"
             >
               {tab.content}
             </div>
           ))}
         </div>
       </div>
       ```

    **Step A — Create `app/debates/[debateId]/MobileLayout.tsx`**:

    ```tsx
    'use client';

    import { HlsPlayer } from './HlsPlayer';
    import { SegmentTimeline } from './SegmentTimeline';
    import { MobileTabs, type MobileTab } from './MobileTabs';

    interface MobileLayoutProps {
      hlsUrl: string | null;
      status: 'live' | 'completed' | 'scheduled';
      topic: string | null;
    }

    export function MobileLayout({ hlsUrl, status, topic }: MobileLayoutProps) {
      const videoNode = (
        <div className="relative w-full bg-black portrait:aspect-video landscape:h-full">
          {hlsUrl ? (
            <HlsPlayer src={hlsUrl} />
          ) : (
            <div className="absolute inset-0 flex items-center justify-center text-sm text-slate-400">
              {status === 'scheduled' ? 'Waiting for debate to start…' : 'Stream unavailable.'}
            </div>
          )}
        </div>
      );

      const tabs: MobileTab[] = [
        {
          id: 'info',
          label: 'Info',
          content: (
            <div className="p-4 space-y-3 text-sm text-slate-200">
              <h2 className="text-base font-semibold">{topic ?? 'Debate'}</h2>
              <p className="text-slate-400">
                This is a live debate.  The segment timeline above shows the current phase.
                Transcript and notes panels will be available in later phases.
              </p>
            </div>
          ),
        },
        {
          id: 'transcript',
          label: 'Transcript',
          content: (
            <div className="p-4 text-sm text-slate-400">
              Transcript — available in a future update (Phase 4).
            </div>
          ),
        },
        {
          id: 'notes',
          label: 'Notes',
          content: (
            <div className="p-4 text-sm text-slate-400">
              Notes — available in a future update (Phase 5).
            </div>
          ),
        },
      ];

      return (
        <div className="portrait:flex portrait:flex-col portrait:h-screen landscape:flex landscape:flex-row landscape:h-screen bg-slate-950 text-slate-100">
          {/* PORTRAIT */}
          <div className="landscape:hidden flex flex-col flex-1 min-h-0">
            {/* Video (full-width) */}
            {videoNode}

            {/* Sticky timeline below video */}
            <div className="sticky top-0 z-10">
              <SegmentTimeline variant="mobile" />
            </div>

            {/* Swipeable tabs fill remaining space */}
            <MobileTabs tabs={tabs} className="flex-1 min-h-0" />
          </div>

          {/* LANDSCAPE */}
          <div className="portrait:hidden landscape:flex landscape:flex-row landscape:flex-1 landscape:min-h-0">
            {/* Video panel ~60% */}
            <div className="landscape:w-3/5 landscape:h-full">
              {videoNode}
            </div>
            {/* Info + timeline panel ~40% */}
            <div className="landscape:w-2/5 landscape:h-full landscape:overflow-y-auto landscape:border-l landscape:border-slate-800 p-4 flex flex-col gap-4">
              <h2 className="text-base font-semibold">{topic ?? 'Debate'}</h2>
              <SegmentTimeline variant="mobile" />
              <p className="text-sm text-slate-400">
                Rotate to portrait for the full multi-tab experience.
              </p>
            </div>
          </div>
        </div>
      );
    }
    ```

    Notes on the above:
    - We use **single-branch** `portrait:hidden` / `landscape:hidden` so the DOM contains both layouts but only the one matching the current orientation is visible.  Avoids layout flash on rotation.
    - HlsPlayer uses `aspect-video` inside its own wrapper.  In landscape we override with `h-full` so video fills the vertical space.
    - The MobileLayout does NOT render a header — in portrait the topic appears in the Info tab; in landscape it appears in the right-hand column.
