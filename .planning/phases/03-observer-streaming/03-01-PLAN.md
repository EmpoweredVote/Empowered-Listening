<objective>
Wire LiveKit RoomCompositeEgress to Mux Live Streaming so that when a moderator starts the first segment
of a debate, the speaker room is composited and pushed via RTMP to Mux, producing an HLS playback URL
that Phase 03-02 will consume.  When the last segment ends (debate transitions to 'completed'), the
pipeline shuts down cleanly and the Mux VOD asset finalizes.

Purpose:
- Unlocks the entire Phase 3 observer streaming track — 03-02/03/04/05 all depend on an HLS stream
  existing for a live debate
- Replaces the stale Cloudflare Stream plumbing implied by OBS-01 (Mux is the actual target per
  project decision)
- Establishes idempotent start/stop semantics so the moderator UI can retry without side effects

Output:
- One Supabase migration (4 new columns on listening.debates)
- One dependency installed (@mux/mux-node)
- env.ts tightened (MUX_TOKEN_ID + MUX_TOKEN_SECRET required)
- Two new server-only lib modules (lib/mux/client.ts, lib/livekit/egress-service.ts)
- One modified API route (segment start/end wires up egress)
- One new API route (GET /api/debates/[debateId]/stream for Phase 03-02)
- Human-verified pipeline: moderator starts segment -> Mux dashboard shows "Active" within 60s
</objective>

    ```ts
    import Mux from '@mux/mux-node';
    import { env } from '@/lib/env';

    let _mux: Mux | null = null;

    function getMux(): Mux {
      if (!_mux) {
        _mux = new Mux({
          tokenId: env.MUX_TOKEN_ID,
          tokenSecret: env.MUX_TOKEN_SECRET,
        });
      }
      return _mux;
    }

    export interface MuxStreamInfo {
      muxStreamId: string;
      muxStreamKey: string;
      muxPlaybackId: string;
    }

    /**
     * Create a new Mux live stream configured for reduced-latency HLS (12-20s delay).
     * reconnect_window: 0 prevents duplicate VOD assets on egress restart (see RESEARCH Pitfall 3).
     */
    export async function createMuxLiveStream(): Promise<MuxStreamInfo> {
      const stream = await getMux().video.liveStreams.create({
        playback_policies: ['public'],
        new_asset_settings: { playback_policies: ['public'] },
        latency_mode: 'reduced',
        reconnect_window: 0,
      });
      const playback = stream.playback_ids?.[0];
      if (!stream.id || !stream.stream_key || !playback?.id) {
        throw new Error('Mux live stream response missing id, stream_key, or playback_ids[0]');
      }
      return {
        muxStreamId:   stream.id,
        muxStreamKey:  stream.stream_key,
        muxPlaybackId: playback.id,
      };
    }

    /** Signal end-of-stream: flush HLS manifest, finalize VOD asset. */
    export async function completeMuxLiveStream(muxStreamId: string): Promise<void> {
      await getMux().video.liveStreams.complete(muxStreamId);
    }
    ```

    File B — `lib/livekit/egress-service.ts`:
    ```ts
    import 'server-only';
    import {
      EgressClient,
      StreamOutput,
      StreamProtocol,
      EncodingOptionsPreset,
    } from 'livekit-server-sdk';
    import { env } from '@/lib/env';

    let _egressClient: EgressClient | null = null;

    function getEgressClient(): EgressClient {
      if (!_egressClient) {
        if (!env.LIVEKIT_URL || !env.LIVEKIT_API_KEY || !env.LIVEKIT_API_SECRET) {
          throw new Error('LiveKit credentials are not configured');
        }
        _egressClient = new EgressClient(
          env.LIVEKIT_URL,
          env.LIVEKIT_API_KEY,
          env.LIVEKIT_API_SECRET,
        );
      }
      return _egressClient;
    }

    /**
     * Start a RoomCompositeEgress pushing RTMP to Mux.
     *
     * Idempotent: if an active egress already exists for this room (same debate, retry),
     * returns its egressId without starting a second job.
     *
     * Uses the LiveKit-native `mux://<stream_key>` shorthand (LiveKit routes this to Mux's
     * RTMP ingest internally).  Layout 'speaker' is LiveKit's built-in speaker-focused composite.
     */
    export async function startDebateEgress(
      roomName: string,
      muxStreamKey: string,
    ): Promise<string> {
      const client = getEgressClient();

      // Idempotency: reuse active egress if one exists.
      const active = await client.listEgress({ roomName, active: true });
      if (active.length > 0 && active[0].egressId) {
        return active[0].egressId;
      }

      const info = await client.startRoomCompositeEgress(
        roomName,
        new StreamOutput({
          protocol: StreamProtocol.RTMP,
          urls: [`mux://${muxStreamKey}`],
        }),
        {
          layout: 'speaker',
          encodingOptions: EncodingOptionsPreset.H264_720P_30,
        },
      );
      if (!info.egressId) {
        throw new Error('startRoomCompositeEgress returned no egressId');
      }
      return info.egressId;
    }

    export async function stopDebateEgress(egressId: string): Promise<void> {
      await getEgressClient().stopEgress(egressId);
    }
    ```

    Do NOT export `getMux` or `getEgressClient` — keep the singletons private so callers are

    ```ts
    import { NextRequest, NextResponse } from 'next/server';
    import { getPool } from '@/lib/db/pool';
    import { requireUserFromRequest } from '@/lib/auth/require-user'; // see note below

    /**
     * GET /api/debates/[debateId]/stream
     *
     * Returns the Mux playback ID + debate status for observer HLS pages (Phase 03-02).
     * NEVER returns mux_stream_key or mux_stream_id — those are server-only secrets.
     */
    export async function GET(
      req: NextRequest,
      { params }: { params: Promise<{ debateId: string }> },
    ) {
      const { debateId } = await params;

      // Auth: any authenticated user may view an observer stream (public debates).
      // If a helper named requireUserFromRequest does not exist in this repo, use the
      // existing pattern from /api/debates/[debateId]/snapshot/route.ts (whichever is
      // the thinnest SSO gate — do NOT require moderator).  Fallback: read ev_token
      // cookie and verify via lib/auth JWKS pattern.
      try {
        await requireUserFromRequest(req);
      } catch {
        return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
      }

      const { rows } = await getPool().query<{
        mux_playback_id: string | null;
        status: string;
      }>(
        `SELECT mux_playback_id, status
         FROM listening.debates WHERE id = $1`,
        [debateId],
      );

      if (rows.length === 0) {
        return NextResponse.json({ error: 'Debate not found' }, { status: 404 });
      }

      // Explicit whitelist of fields — no accidental leakage of mux_stream_key.
      return NextResponse.json({
        mux_playback_id: rows[0].mux_playback_id,
        status: rows[0].status,
      });
    }
    ```
