**1. `lib/auth/jwks.ts`** — Shared JWKS instance and `verifyToken` helper using `jose`.

```typescript
import { jwtVerify, createRemoteJWKSet, type JWTPayload } from 'jose';

const JWKS_URL = new URL(
  process.env.JWKS_URL ?? 'https://kxsdzaojfaibhuzmclfq.supabase.co/auth/v1/.well-known/jwks.json'
);

export const JWKS = createRemoteJWKSet(JWKS_URL);

export async function verifyToken(token: string): Promise<JWTPayload> {
  const { payload } = await jwtVerify(token, JWKS, {
    issuer: 'https://kxsdzaojfaibhuzmclfq.supabase.co/auth/v1',
    audience: 'authenticated',
    algorithms: ['ES256'],
  });
  return payload;
}
```

Use `jose` (NOT `jsonwebtoken` — that library uses Node crypto and fails on Workers).  The `algorithms: ['ES256']` option is required — without it jose will accept any algorithm in the JWKS, including symmetric ones.

**2. `lib/auth/session.ts`** — Silent session renewal.

```typescript
export async function tryRenewSession(): Promise<string | null> {
  try {
    const res = await fetch('https://api.empowered.vote/api/auth/session', {
      credentials: 'include',
    });
    if (!res.ok) return null;
    const { access_token } = await res.json();
    return access_token;
  } catch {
    return null;
  }
}
```

`credentials: 'include'` is load-bearing.  Without it the `ev_session` httpOnly cookie never crosses origins and silent renewal always returns null.

**6. `middleware.ts`** at the project root.

```typescript
import { NextRequest, NextResponse, userAgent } from 'next/server';
import { verifyToken } from './lib/auth/jwks';

const JOIN_PATHS = /^\/join\/(speaker|moderator)(\/|$)/;

export async function middleware(request: NextRequest) {
  const pathname = request.nextUrl.pathname;
  const isJoinPath = JOIN_PATHS.test(pathname);

  // Desktop gate: applies only to /join/speaker/* and /join/moderator/*
  if (isJoinPath) {
    const { device } = userAgent(request);
    const isMobileOrTablet = device.type === 'mobile' || device.type === 'tablet';
    if (isMobileOrTablet) {
      // Signal to the page component to render the gate inline.  No redirect.
      const response = NextResponse.next();
      response.headers.set('x-mobile-gate', '1');
      return response;
    }
  }

  // Bypass in local dev short-circuits JWT verification entirely.
  if (process.env.AUTH_BYPASS === '1' && process.env.NODE_ENV === 'development') {
    const response = NextResponse.next();
    response.headers.set('x-user-id', '00000000-0000-0000-0000-000000000001');
    return response;
  }

  // JWT verification on protected paths.
  if (isJoinPath) {
    // IMPORTANT: use request.cookies, NOT cookies() from next/headers — Workers runtime.
    const token = request.cookies.get('ev_token')?.value
      ?? request.headers.get('authorization')?.slice(7);

    if (!token) {
      const returnUrl = encodeURIComponent(request.url);
      return NextResponse.redirect(
        `https://accounts.empowered.vote/login?redirect=${returnUrl}`
      );
    }

    try {
      const payload = await verifyToken(token);
      const response = NextResponse.next({
        request: { headers: new Headers(request.headers) },
      });
      response.headers.set('x-user-id', payload.sub as string);
      return response;
    } catch {
      const returnUrl = encodeURIComponent(request.url);
      return NextResponse.redirect(
        `https://accounts.empowered.vote/login?redirect=${returnUrl}`
      );
    }
  }

  return NextResponse.next();
}

export const config = {
  // Apply to join paths only.  The holding page (/) handles its own unauthenticated state client-side.
  matcher: ['/join/:path*'],
};
```

Critical do-not items:
- Do NOT import `cookies` from `next/headers` in middleware.ts (Node.js-only, fails on Workers).
- Do NOT use `jsonwebtoken` — incompatible with Workers runtime.
- Do NOT allow symmetric algorithms — `algorithms: ['ES256']` is required.
- Do NOT call `getAccountMe` from middleware — middleware runs before rendering and should not make outbound API calls per-request.  Account standing check happens in page server components using the `x-user-id` header forwarded by middleware.

**3. `components/auth/SessionProvider.tsx`** — Client provider that attempts silent renewal on mount, parses the hash fragment after SSO return, stores the token in localStorage, and exposes user state via context.

```typescript
'use client';

import { createContext, useContext, useEffect, useState, type ReactNode } from 'react';
import { tryRenewSession } from '@/lib/auth/session';

type SessionState = {
  token: string | null;
  displayName: string | null;
  loading: boolean;
};

const SessionContext = createContext<SessionState>({ token: null, displayName: null, loading: true });

export function useSession() {
  return useContext(SessionContext);
}

export function SessionProvider({ children }: { children: ReactNode }) {
  const [state, setState] = useState<SessionState>({ token: null, displayName: null, loading: true });

  useEffect(() => {
    async function init() {
      // 1. Check for hash fragment from SSO return: #access_token=...
      if (window.location.hash.includes('access_token=')) {
        const params = new URLSearchParams(window.location.hash.slice(1));
        const hashToken = params.get('access_token');
        if (hashToken) {
          localStorage.setItem('ev_token', hashToken);
          // Clear hash from URL bar.
          history.replaceState(null, '', window.location.pathname + window.location.search);
        }
      }

      // 2. Try existing token.
      let token = localStorage.getItem('ev_token');

      // 3. If no token, try silent renewal.
      if (!token) {
        token = await tryRenewSession();
        if (token) localStorage.setItem('ev_token', token);
      }

      // 4. If token, fetch display name.
      let displayName: string | null = null;
      if (token) {
        try {
          const res = await fetch('https://api.empowered.vote/api/account/me', {
            headers: { Authorization: `Bearer ${token}` },
            credentials: 'include',
          });
          if (res.ok) {
            const account = await res.json();
            displayName = account.display_name;
          } else if (res.status === 401) {
            // Token rejected; clear and leave unauthenticated.
            localStorage.removeItem('ev_token');
            token = null;
          }
        } catch {
          // Network error; leave as-is.
        }
      }

      setState({ token, displayName, loading: false });
    }
    init();
  }, []);

  return <SessionContext.Provider value={state}>{children}</SessionContext.Provider>;
}
```

This is client-only and lightweight; server-side auth enforcement happens in middleware for protected routes.
