  <behavior>
    computeDebateTimeMmss(actualStart: Date, spokenAt: Date): string

    Returns elapsed debate time formatted as "mm:ss" (zero-padded both parts).
    Uses wall-clock difference — NOT Deepgram word.start timestamps (which reset on reconnect).

    Test cases (RED first — write these before implementing):
      - actualStart = T, spokenAt = T + 0ms           → "00:00"
      - actualStart = T, spokenAt = T + 1000ms         → "00:01"
      - actualStart = T, spokenAt = T + 90000ms        → "01:30"
      - actualStart = T, spokenAt = T + 150000ms       → "02:30"
      - actualStart = T, spokenAt = T + 3600000ms      → "60:00"
      - spokenAt before actualStart (clock skew guard) → "00:00" (Math.max(0, elapsed))
  </behavior>
  <implementation>
    export function computeDebateTimeMmss(actualStart: Date, spokenAt: Date): string {
      const elapsedMs = spokenAt.getTime() - actualStart.getTime();
      const totalSeconds = Math.floor(Math.max(0, elapsedMs) / 1000);
      const mm = Math.floor(totalSeconds / 60);
      const ss = totalSeconds % 60;
      return `${String(mm).padStart(2, '0')}:${String(ss).padStart(2, '0')}`;
    }
  </implementation>
</feature>
